üîß Mapbox GL JS Styling Example:

// Use the color properties in your map
map.addLayer({
  'id': 'regions',
  'type': 'fill',
  'source': 'regions',
  'paint': {
    // Use the color properties we added
    'fill-color': ['get', 'fill'],
    'fill-opacity': ['get', 'fill-opacity'],
    'fill-outline-color': ['get', 'stroke']
  }
});

// Or style based on region names (counties/dioceses)
map.addLayer({
  'id': 'regions-dynamic',
  'type': 'fill', 
  'source': 'regions',
  'paint': {
    'fill-color': [
      'match',
      ['get', 'name'], // or 'COUNTY', 'diocese', etc.
      'Dublin', '#8b0000',
      'Cork', '#32cd32', 
      'Galway', '#9370db',
      'Kerry', '#ffd700',
      'Mayo', '#ffb6c1',
      // Add more region mappings...
      '#64748b' // default color
    ],
    'fill-opacity': 0.7,
    'fill-outline-color': '#1e293b'
  }
});

// Hover effects
map.on('mouseenter', 'regions', (e) => {
  map.getCanvas().style.cursor = 'pointer';
  
  // Show popup with region name
  new mapboxgl.Popup()
    .setLngLat(e.lngLat)
    .setHTML(`${e.features[0].properties.name}`)
    .addTo(map);
});
