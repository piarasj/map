# Simple jq merge
jq -s '.[0].features + .[1].features | {type: "FeatureCollection", features: .}' ulsterROI.geojson ulsterNI.geojson > ulster-combined.geojson

# Preserve CRS and metadata
jq -s '{
  type: "FeatureCollection",
  name: "Ulster Combined (ROI + NI)",
  crs: .[0].crs,
  features: (.[0].features + .[1].features)
}' ulsterROI.geojson ulsterNI.geojson > ulster-combined.geojson

# Node.js merge script
const fs = require('fs');

const roi = JSON.parse(fs.readFileSync('ulsterROI.geojson', 'utf8'));
const ni = JSON.parse(fs.readFileSync('ulsterNI.geojson', 'utf8'));

const combined = {
  type: 'FeatureCollection',
  name: 'Ulster Combined (ROI + NI)',
  crs: roi.crs || ni.crs,
  features: [...roi.features, ...ni.features]
};

fs.writeFileSync('ulster-combined.geojson', JSON.stringify(combined, null, 2));
console.log(`‚úÖ Merged ${combined.features.length} features`);
