// Use the color properties in your MapaLister app
map.addLayer({
  'id': 'ulster-counties',
  'type': 'fill',
  'source': 'ulster',
  'paint': {
    // Use the color property we added
    'fill-color': ['get', 'fill'],
    'fill-opacity': ['get', 'fill-opacity'],
    'fill-outline-color': ['get', 'stroke']
  }
});

// Or style based on county names
map.addLayer({
  'id': 'ulster-counties-dynamic',
  'type': 'fill', 
  'source': 'ulster',
  'paint': {
    'fill-color': [
      'match',
      ['get', 'COUNTY'], // or 'CountyName' for NI data
      'DONEGAL', '#e11d48',
      'ANTRIM', '#3b82f6',
      'TYRONE', '#10b981',
      'FERMANAGH', '#f59e0b',

      'MONAGHAN', '#84cc16',
      'DOWN', '#f97316',
      '#64748b' // default
    ],
    'fill-opacity': 0.7,
    'fill-outline-color': '#1e293b'
  }
